#include <iostream>
#include <string>
using namespace std;
class Vehicle {
	public:
    void Wheels() {
      cout<<"Depends on the Vehicle!\n";
    }
    void Fly() {
      cout<<"Can I Fly??? Are you Kidding!\n";
    }  
};
class Car: public Vehicle {
	public:
    void Wheels() {
      cout<<"It is 4 Wheels!\n";
    }
};
class Bike: public Vehicle {
	public:
    void Wheels() {
      cout<<"It is 2 Wheels!\n";
    }
};
class Boeing: public Vehicle {
	public:
    void Wheels() {
      cout<<"Too many wheels to count!\n";
    }
    void Fly() {
      cout<<"I can fly but not sure about landing safely!!!\n";
    }   
};

int main() {
  	cout<<"Lets try Car object...\n";
    Car myCar;
    myCar.Wheels();
    myCar.Fly();
  	cout<<"**********************************************\n";
  	cout<<"Now Lets try Bike object...\n";  
    Bike myBike;
    myBike.Wheels();
    myBike.Fly();    
  	cout<<"**********************************************\n";
  	cout<<"Now Lets try Boeing object...\n";  
    Boeing myFlight;
    myFlight.Wheels();
    myFlight.Fly();      
  	cout<<"**********************************************\n";
	return 0;
}
