<script>
class Vehicle {
  Wheels() {
    return "Depends on the Vehicle! <br>";
  }
  Fly() {
    return "Can I Fly??? Are you Kidding! <br>";
  }  
};
class Car extends Vehicle {
  Wheels() {
    return "It is 4 Wheels! <br>";
  }
};
class Bike extends Vehicle {
  Wheels() {
    return "It is 2 Wheels! <br>";
  }
};
class Boeing extends Vehicle {
  Wheels() {
    return "Too many wheels to count! <br>";
  }
  Fly() {
    return "I can fly but not sure about landing safely!!! <br>";
  }   
};

let myCar = new Car();
document.getElementById("carwheel").innerHTML = myCar.Wheels();    
document.getElementById("carfly").innerHTML = myCar.Fly();

let myBike = new Bike();
document.getElementById("bikewheel").innerHTML = myBike.Wheels();    
document.getElementById("bikefly").innerHTML = myBike.Fly();

let myBoeing = new Boeing();
document.getElementById("boeingwheel").innerHTML = myBoeing.Wheels();    
document.getElementById("boeingfly").innerHTML = myBoeing.Fly();
</script>
